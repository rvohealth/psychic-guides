# generating

To generate a serializer, use the cli command provided:

```bash
yarn psy g:serializer user email:string id:string created_at:datetime
```

```ts
import { DateTime } from 'luxon'
import { Attribute, RendersOne, DreamSerializer } from '@rvohealth/dream'

export default class UserSerializer<
  DataType extends User,
> extends DreamSerializer<DataType> {
  @Attribute('string')
  public email: string

  @Attribute('string')
  public id: string

  @Attribute()
  public createdAt: DateTime
}
```
