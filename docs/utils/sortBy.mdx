# sortBy

Returns a copy of the array sorted by the return value of the function passed as the second argument.

```ts
import { sortBy } from '@rvoh/dream'

sortBy(['aaa', 'a', 'aa'], (str: string) => str.length)
// ['a', 'aa', 'aaa']

sortBy(['aaa', 'a', 'aa'], (str: string) => -str.length)
// ['aaa', 'aa', 'a']

sortBy([5, 3, 7], (num: number) => num)
// [3, 5, 7]

sortBy([5, 3, 7], (num: number) => -num)
// [7, 5, 3]
```
